java+lombok 带有中文注释的/***/注释方式的课程管理实体类
/**
 * 课程管理实体类
 */
@Data
@NoArgsConstructor
@AllArgsConstructor
public class Course {
    
    /**
     * 课程编号
     */
    private Long id;
    
    /**
     * 课程名称
     */
    private String name;
    
    /**
     * 课程描述
     */
    private String description;
    
    /**
     * 课程开始时间
     */
    private Date startTime;
    
    /**
     * 课程结束时间
     */
    private Date endTime;
    
    /**
     * 课程状态
     * 0-未开始
     * 1-进行中
     * 2-已结束
     */
    private Integer status;
    
}
